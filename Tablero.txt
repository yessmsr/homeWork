from Dado import *
from Ficha import *


class Tablero:
    casillas=20
    fichas=[]

    ##################
    posicion=Ficha(0)
    ficha1=Ficha("Verde")
    ficha2=Ficha("Azul")
    def __init__(self, casillas, fichas):
        self.casillas=20
        self.fichas=fichas
    
    def juego(self):
        casilla=self.casillas
        while (self.ficha1<=casilla and self.ficha2<=casilla):
            avanza=self.posicion.avanzar()
        # Lanzamiendo Jugador1
            self.ficha1=avanza
            self.ficha2=avanza
            if (self.ficha1>=casilla):
                print("Ha ganado la ficha verde")

        # Lanzamiendo Jugador2
            if (self.ficha2>=casilla):
                print("Ha ganado la ficha Azul")
